#+title: nginx站点跳转

从dirlt.com跳转到dirtysalt.info
#+BEGIN_EXAMPLE
http {
    server_name dirlt.com;
    rewrite ^ $scheme://dirtysalt.info$request_uri? permanent;
}
#+END_EXAMPLE
